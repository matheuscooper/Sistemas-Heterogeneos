#include <fstream>
using namespace std;

int main() {
    ofstream image("images/Imagem.ppm");

    if (!image.is_open()) {
        cerr << "Erro ao criar o arquivo!" << endl;
        return 1;
    }

    int width = 70;
    int height = 50;

    // CabeÃ§alho do PPM (P3 = imagem em ASCII)
    image << "P3\n";
    image << width << " " << height << "\n";
    image << "255\n"; // MÃ¡ximo valor de cor

    // Gera a imagem linha por linha
    for (int y = 0; y < height; y++) {
        for (int x = 0; x < width; x++) {
            if (x < width / 2) {
                image << "255 0 0 "; // Vermelho
            } else {
                image << "0 0 255 "; // Azul
            }
        }
        image << "\n";
    }

    image.close();
    cout << "Imagem gerada com sucesso!" << endl;
